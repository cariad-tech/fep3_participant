find_package(dev_essential REQUIRED COMPONENTS result)

set(TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET test_component_intfs_job_reg_c_plugin)

set(BUILD_INTERFACE_INCLUDES
    ${PROJECT_SOURCE_DIR}/include
    ${PROJECT_BINARY_DIR}/include
)

add_library(${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET} SHARED
    ${PROJECT_SOURCE_DIR}/src/fep3/components/job_registry/mock/mock_job.h
    ${PROJECT_SOURCE_DIR}/src/fep3/components/job_registry/mock/mock_job_registry.h
    test_plugin/job_registry.cpp
    test_plugin/test_plugin.h
    test_plugin/test_plugin.cpp
)
# some include files are generated by fep3_participant so the plugin depends on fep3_participant
add_dependencies(${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET} fep3_participant)
target_include_directories(${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET} PRIVATE
    "$<BUILD_INTERFACE:${BUILD_INTERFACE_INCLUDES}>"
    "$<INSTALL_INTERFACE:${INSTALL_INTERFACE_INCLUDES}>"
)
target_link_libraries(${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET} PRIVATE dev_essential::result)
target_link_libraries(${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET} PRIVATE
    GTest::gtest_main
    GTest::gmock
)
target_compile_definitions(${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET} PRIVATE _FEP3_PARTICIPANT_INCLUDED_STATIC)
set_target_properties(${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET} PROPERTIES FOLDER "test/private/component_intfs/job_registry/c/test_plugin")

##################################################################
#  test application for Job Registry interface of the C Plugin System
##################################################################

add_definitions(-DCURRENT_TEST_DIR="${CMAKE_CURRENT_SOURCE_DIR}/../")
add_definitions(-DTEST_PLUGIN_DIR="${CMAKE_CURRENT_BINARY_DIR}")

add_executable(tester_component_intfs_job_reg_c_plugin
    tester_job_reg_intf_c_plugin.cpp
)
target_include_directories(tester_component_intfs_job_reg_c_plugin PRIVATE
)
target_compile_definitions(tester_component_intfs_job_reg_c_plugin PRIVATE
    PLUGIN="$<TARGET_FILE:${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET}>"
    TEST_BUILD_DIR="$<TARGET_FILE_DIR:tester_component_intfs_job_reg_c_plugin>"
)
target_link_libraries(tester_component_intfs_job_reg_c_plugin PRIVATE
    GTest::gmock_main
    ${CMAKE_DL_LIBS}
    fep3_participant_private_lib
    participant_private_test_utils
)
target_link_libraries(tester_component_intfs_job_reg_c_plugin PUBLIC
    dev_essential::result
)

add_test(NAME tester_component_intfs_job_reg_c_plugin
    COMMAND tester_component_intfs_job_reg_c_plugin
    TIMEOUT 10
)
set_target_properties(tester_component_intfs_job_reg_c_plugin PROPERTIES FOLDER "test/private/component_intfs/job_registry/c")

##################################################################
#  test application for Job Registry RPC implementation of the C Plugin System
##################################################################

jsonrpc_generate_client_stub(${PROJECT_SOURCE_DIR}/include/fep3/rpc_services/job_registry/job_registry.json
                             test::rpc_stubs::TestJobRegistryClientStub
                             ${CMAKE_CURRENT_BINARY_DIR}/test_job_registry_client_stub.h)

add_executable(tester_rpc_implementation_job_reg_c_plugin
    tester_job_reg_rpc_c_plugin.cpp
    ${CMAKE_CURRENT_BINARY_DIR}/test_job_registry_client_stub.h
)
target_include_directories(tester_rpc_implementation_job_reg_c_plugin PRIVATE
    ${CMAKE_CURRENT_BINARY_DIR}
)
target_compile_definitions(tester_rpc_implementation_job_reg_c_plugin PRIVATE
    PLUGIN="$<TARGET_FILE:${TEST_COMPONENT_INTERFACES_JOB_REGISTRY_C_PLUGIN_TARGET}>"
    TEST_BUILD_DIR="$<TARGET_FILE_DIR:tester_rpc_implementation_job_reg_c_plugin>"
)
target_link_libraries(tester_rpc_implementation_job_reg_c_plugin PRIVATE
    GTest::gmock_main
    ${CMAKE_DL_LIBS}
    fep3_participant_private_lib
    participant_private_test_utils
)
target_link_libraries(tester_rpc_implementation_job_reg_c_plugin PUBLIC
    dev_essential::result
)

add_test(NAME tester_rpc_implementation_job_reg_c_plugin
    COMMAND tester_rpc_implementation_job_reg_c_plugin
    TIMEOUT 10
)
set_target_properties(tester_rpc_implementation_job_reg_c_plugin PROPERTIES FOLDER "test/private/component_intfs/job_registry/c")
